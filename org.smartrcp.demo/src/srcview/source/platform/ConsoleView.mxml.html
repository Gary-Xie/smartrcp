<?xml version="1.0" encoding="utf-8"?>
<gui:RCPModule xmlns:mx="http://www.adobe.com/2006/mxml" 
    xmlns:ui="platform.*"
    xmlns:gui="cn.smartinvoke.gui.*" xmlns:rcp="cn.smartinvoke.rcp.*"
    layout="vertical" creationComplete="init()">
    <mx:Script>
        <![CDATA[
            import cn.smartinvoke.smartrcp.gui.module.CActionEvent;
            import mx.collections.ArrayCollection;
            import cn.smartinvoke.rcp.CToolBar;
            import cn.smartinvoke.smartrcp.gui.FlashViewPart;
            import cn.smartinvoke.RemoteObject;
            import cn.smartinvoke.smartrcp.gui.ViewPartActionBar;
            import cn.smartinvoke.smartrcp.ConsoleServer;
            var consoleService:ConsoleServer;
            function init():void{
                this.consoleService=new ConsoleServer();
                this.consoleService.addListener(function (ret:Array):void{
                     var isError:Boolean=ret[0] as Boolean;
                     if(isError){
                        var msg:String='<font color="#FF0000">'+ret[1]+'</font>';
                        this.consoleView.htmlText=this.consoleView.htmlText+msg;
                     }else{
                        this.consoleView.htmlText=this.consoleView.htmlText+ret[1];
                     }
                },this);
                
                this.initActionBars();
            }
            
            var actionBar:ViewPartActionBar=null;
            function initActionBars():void{  
                var parent:RemoteObject=this.flashViewer.getParent();
                //如果存载当前flash的容器时viewPart，才具有此功能
                if(parent is FlashViewPart){
                   var viewPart:FlashViewPart=parent as FlashViewPart;
                   this.actionBar=viewPart.getPartActionBar();
                   this.actionBar.addAction(this.actionClear);
                   
                   var toolBar:CToolBar=new CToolBar();
                   toolBar.actionIds=new ArrayCollection([this.actionClear.actionId]);
                   
                   this.actionBar.fillToolBar(toolBar);
                   
                   this.actionBar.addListener(this.actionClear.actionId,function (evt:CActionEvent):void{
                         this.consoleView.htmlText="";
                   },this);
                }
            }
        ]]>
    </mx:Script>
    <rcp:CAction id="actionClear" actionId="clearAction" 
        imageUrl="icons/clear.gif" text="清空控制台"/>
    <ui:ConsoleTextArea id="consoleView" width="100%" height="100%">
        
    </ui:ConsoleTextArea>
</gui:RCPModule>
